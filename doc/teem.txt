CREATE TABLE `pl_bid_info` (
`bid_id` varchar(100) NOT NULL COMMENT '申请ID',
`user_id` varchar(20) DEFAULT NULL COMMENT '申请人ID',
`bid_name` varchar(100) DEFAULT '' COMMENT '标书名称',
`bid_number` varchar(100) DEFAULT '' COMMENT '标书编号',
`bid_deadline` date DEFAULT NULL COMMENT '投标截止日期',
`bid_date` datetime DEFAULT NULL COMMENT '投标日期',
`winning_date` date DEFAULT NULL COMMENT '中标日期',
`bid_opening_date` date DEFAULT NULL COMMENT '开标日期',
`tender_agency` varchar(100) DEFAULT '' COMMENT '招标代理公司',
`bidding_unit` varchar(100) DEFAULT '' COMMENT '招标单位',
`quoted_price` decimal(10,2) DEFAULT '0.00' COMMENT '报价(元)',
`construction_site` varchar(100) DEFAULT '' COMMENT '建设地点',
`bid_bond` decimal(10,2) DEFAULT '0.00' COMMENT '投标保证金(元)',
`win_bid` char(1) DEFAULT '0' COMMENT '是否中标(0否 1是)',
`bid_bond_recovered` varchar(100) DEFAULT '0' COMMENT '投标保证金是否收回(0否 1是)',
`bid_bond_takeback_date` datetime DEFAULT NULL COMMENT '投标保证金收回日期',
`bid_bond_reason_for_failure` varchar(500) DEFAULT '' COMMENT '投标保证金未收回原因',
`perform_bond` decimal(10,2) DEFAULT NULL COMMENT '履约保证金(元)',
`perform_bond_recovered` char(1) DEFAULT '0' COMMENT '履约保证金是否收回(0否 1是)',
`perform_bond_takeback_date` datetime DEFAULT NULL COMMENT '履约保证金收回日期',
`perform_bond_reason_for_failure` varchar(500) DEFAULT '' COMMENT '履约保证金未收回原因',
`bid_attachment` varchar(2000) DEFAULT '' COMMENT '中标通知书附件信息(前端自主构建JSON格式字符串)',
`instance_id` varchar(100) DEFAULT '' COMMENT '流程实例ID',
`status` char(1) DEFAULT '0' COMMENT '状态（0正常 1停用）',
`apply_status` char(1) DEFAULT '0' COMMENT '申请状态(0-未提交 1-审批中 2-被退回 3-审批通过 4-审批未通过 5-已撤销)',
`del_flag` char(1) DEFAULT '0' COMMENT '删除标识',
`create_by` varchar(64) DEFAULT '' COMMENT '创建者',
`create_time` datetime DEFAULT NULL COMMENT '创建时间',
`update_by` varchar(64) DEFAULT '' COMMENT '更新者',
`update_time` datetime DEFAULT NULL COMMENT '更新时间',
`remark` varchar(500) DEFAULT NULL COMMENT '备注',
`apply_time` datetime DEFAULT NULL COMMENT '申请时间',
 `project_id` varchar(100) DEFAULT '' COMMENT '投标项目',
`project_category` varchar(200) DEFAULT NULL COMMENT '项目类别',
`project_manager` varchar(200) DEFAULT NULL COMMENT '项目负责人',
`project_manager_name` varchar(100) DEFAULT NULL COMMENT '项目负责人姓名',
`project_manager_contact` varchar(200) DEFAULT NULL COMMENT '项目负责人电话',
`agent` varchar(200) DEFAULT NULL COMMENT '经办人',
`agent_name` varchar(100) DEFAULT NULL COMMENT '经办人姓名',
`agent_contact` varchar(200) DEFAULT NULL COMMENT '经办人电话',
`project_type` varchar(200) DEFAULT NULL COMMENT '项目类型',
`scale` varchar(200) DEFAULT NULL COMMENT '规模',
`business_category` varchar(200) DEFAULT NULL COMMENT '业务类别',
`union_company_name` varchar(200) DEFAULT NULL COMMENT '联合单位名称',
`bid_place` varchar(200) DEFAULT NULL COMMENT '投标地点',
PRIMARY KEY (`apply_id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='投标信息申请表'
